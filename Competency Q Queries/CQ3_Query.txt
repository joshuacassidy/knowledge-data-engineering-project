#Q3. What are the top host acceptance rate with their IDs in Ireland where the activity is high in ‘Clare’ and ‘Dublin’ county?
PREFIX host_acceptancerate: <http://www.semanticweb.org/ontologies/2020/10/Host>
PREFIX Activity:<http://www.semanticweb.org/ontologies/2020/10/Activity>
PREFIX owl:<http://www.w3.org/2002/07/owl#>
PREFIX rdfs:<http://www.w3.org/2007/01/rdf-schema#>
PREFIX xsd:<http://www.w3.org/2001/XMLSchema#>
SELECT DISTINCT ?hostID ?acceptanceRate ?ActivityLocation WHERE{      
    ?hostID <http://www.semanticweb.org/ontologies/2020/10/hostAcceptanceRate> ?acceptanceRate.
    ?location  <http://www.semanticweb.org/ontologies/2020/10/activityLocation> ?ActivityLocation
    FILTER (?acceptanceRate > "90" && ?ActivityLocation in ("Clare", "Dublin"))
}
GROUP BY ?acceptanceRate ?hostID ?ActivityLocation ORDER BY DESC(?acceptanceRate)
LIMIT 100
