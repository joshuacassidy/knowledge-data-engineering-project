#Q8. Where are the Top Accommodation Rating over all counties in Ireland. 
PREFIX AccRating:<http://www.semanticweb.org/ontologies/2020/10/AccommodationRating>
PREFIX AccName: <http://www.semanticweb.org/ontologies/2020/10/Accomedation>
PREFIX Activity:<http://www.semanticweb.org/ontologies/2020/10/Activity>
PREFIX owl:<http://www.w3.org/2002/07/owl#>
PREFIX rdfs:<http://www.w3.org/2007/01/rdf-schema#>
PREFIX xsd:<http://www.w3.org/2001/XMLSchema#>
SELECT DISTINCT ?AccomodationRating ?ActivityLocation {
    ?rating <http://www.semanticweb.org/ontologies/2020/10/accRating> ?AccomodationRating .    

    ?location  <http://www.semanticweb.org/ontologies/2020/10/activityLocation> ?ActivityLocation .
    FILTER (?AccomodationRating > "80")
}
GROUP BY ?ActivityLocation ?AccomodationRating ORDER BY ASC(?AccomodationRating)
LIMIT 111
