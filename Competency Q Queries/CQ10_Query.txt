#Q10. How many Airbnbs in Dublin have the price higher than 100Eu from most tourists spots?
PREFIX Availability:<http://www.semanticweb.org/ontologies/2020/10/Availability>
PREFIX MostActivity:<http://www.semanticweb.org/ontologies/2020/10/Activity>
PREFIX owl:<http://www.w3.org/2002/07/owl#>
PREFIX rdfs:<http://www.w3.org/2007/01/rdf-schema#>
PREFIX xsd:<http://www.w3.org/2001/XMLSchema#>
SELECT DISTINCT ?ActivityName ?airbnbName ?AvailablePrice WHERE{
    ?price <http://www.semanticweb.org/ontologies/2020/10/availabilityPrice> ?AvailablePrice .
    ?s <http://www.semanticweb.org/ontologies/2020/10/activityLocation> "Dublin".
    ?airbnblocation <http://www.semanticweb.org/ontologies/2020/10/accommodationLocatedIn>
    <http://foo.example/Location/Dublin>.
    ?airbnbName <http://www.semanticweb.org/ontologies/2020/10/accName> ?ActivityName
    FILTER (?AvailablePrice > "100")
}
#GROUP BY ?ActivityName ?airbnbName ?AvailablePrice ORDER BY ASC(?AvailablePrice)
LIMIT 250
