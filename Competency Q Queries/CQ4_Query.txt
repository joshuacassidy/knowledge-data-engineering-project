#Q4. Besides ‘Dublin’ tourist location, list the owners of the Airbnbs? 
PREFIX SuperHost:<http://www.semanticweb.org/ontologies/2020/10/Host>
PREFIX Activity: <http://www.semanticweb.org/ontologies/2020/10/Activity>
PREFIX owl:<http://www.w3.org/2002/07/owl#>
PREFIX rdfs:<http://www.w3.org/2007/01/rdf-schema#>
PREFIX xsd:<http://www.w3.org/2001/XMLSchema#>
SELECT DISTINCT ?SuperHost ?airbnblocation ?ActivityLocation WHERE{
    ?s <http://www.semanticweb.org/ontologies/2020/10/activityLocation> ?c .
    ?airbnblocation <http://www.semanticweb.org/ontologies/2020/10/accommodationLocatedIn> ?accLoc .
    ?airbnbName <http://www.semanticweb.org/ontologies/2020/10/isSuperHost> ?SuperHost.
    ?s  <http://www.semanticweb.org/ontologies/2020/10/activityLocation> ?ActivityLocation
    FILTER (?SuperHost = "true" && (?ActivityLocation != 'Dublin'))
}
GROUP BY ?SuperHost ?airbnblocation ?ActivityLocation ORDER BY (?ActivityLocation)
