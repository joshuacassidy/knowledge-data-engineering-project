#Q7.In Ireland how many Airbnbs RoomTypes are avalable around for Tourists 
PREFIX RoomType:<http://www.semanticweb.org/ontologies/2020/10/Activity>
PREFIX owl:<http://www.w3.org/2002/07/owl#>
PREFIX rdfs:<http://www.w3.org/2007/01/rdf-schema#>
PREFIX xsd:<http://www.w3.org/2001/XMLSchema#>
SELECT ?RoomTypeID ?activityLocation ?RoomTypeName WHERE{
    ?s <http://www.semanticweb.org/ontologies/2020/10/activityLocatedIn>
    ?activityLocation.
    ?RoomTypeID <http://www.semanticweb.org/ontologies/2020/10/roomType> ?RoomTypeName .
}
GROUP BY ?RoomTypeID ?activityLocation ?RoomTypeName ORDER BY ?RoomTypeName
LIMIT 75
